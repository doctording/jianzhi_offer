 public class ListNode {
    int val;

public class Solution {

    public ListNode EntryNodeOfLoop(ListNode pHead)
    {
        if(pHead == null || pHead.next == null){
            return null;
        }
        // 都从pHead开始
        ListNode p1 = pHead;
        ListNode p2 = pHead;
        while(p1 != null || p2 != null){
            // 走一步
            p1 = p1.next;
            if(p1 == null){
                break;
            }
            // 走两步
            p2 = p2.next;
            if(p2 == null){
                break;
            }
            p2 = p2.next;
            if(p1 == p2){
                ListNode t1 = pHead;
                ListNode t2 = p1;
                while(t1 != t2){
                    t1 = t1.next;
                    t2 = t2.next;
                }
                return t1;
            }
        }
        return null;
    }
}
