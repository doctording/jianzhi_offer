import java.util.ArrayList;
import java.util.List;

public class Solution {
    public int GetUglyNumber_Solution(int index) {
        List<Integer> li = new ArrayList(512);
        if(index == 0){
            return 0;
        }
        int pos = index - 1;
        li.add(1);
        int i2 = 0;
        int i3 = 0;
        int i5 = 0;
        while((index --) > 0){
            int v2 = 2 * li.get(i2);
            int v3 = 3 * li.get(i3);
            int v5 = 5 * li.get(i5);
            int minVal = Math.min(v2, v3);
            minVal = Math.min(minVal,v5);

            li.add(minVal);

            if(v2 <= minVal){
                i2 ++;
            }
            if(v3 <= minVal){
                i3 ++;
            }
            if(v5 <= minVal){
                i5 ++;
            }
        }
        return li.get(pos);
    }
}
