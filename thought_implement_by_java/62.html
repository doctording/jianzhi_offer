import java.util.*;

    }

public class Solution {
    TreeNode KthNode(TreeNode pRoot, int k)
    {
        if(pRoot == null){
            return null;
        }
        int i = 0;
        Stack<TreeNode> sta = new Stack<>();
        sta.push(pRoot);
        TreeNode p = pRoot.left;
        while(p != null || !sta.isEmpty()){
            while(p != null){
                sta.push(p);
                p = p.left;
            }
            TreeNode popNode = sta.pop();
            i++;
            if(i == k){
                return popNode;
            }
//            System.out.print(popNode.val + " ");
            if(popNode.right != null){
                p = popNode.right;
            }
        }
        return null;
    }

}

import java.util.*;
/*
public class TreeNode {
    int val = 0;
    TreeNode left = null;
    TreeNode right = null;

    public TreeNode(int val) {
        this.val = val;

    }

}
*/
public class Solution {
    TreeNode KthNode(TreeNode pRoot, int k)
    {
        if(pRoot == null){
            return null;
        }
        TreeNode cur = pRoot;
        int i = 0;
        while(cur != null){
            if(cur.left == null){ // 最左
//                System.out.print(cur.val + " ");
                i++;
                if(i == k){
                    return cur;
                }
                cur = cur.right;
            }else{
                TreeNode preCur = cur.left;
                while(preCur.right != null && preCur.right != cur){
                    preCur = preCur.right;
                }
                if(preCur.right == null){ // 第一次访问
                    preCur.right = cur;
                    cur = cur.left;
                }else {
//                    System.out.print(cur.val + " "); // 第二次访问
                    i++;
                    if(i == k){
                        return cur;
                    }
                    preCur.right = null;
                    cur = cur.right;
                }
            }
        }
        return null;
    }

}
