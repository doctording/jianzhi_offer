28 数组中出现次数超过一半的数字

public class Solution {
     public int MoreThanHalfNum_Solution(int [] array) {
        int n = array.length;
        if(n <= 0){
            return 0;
        }
        int candidate = array[0];
        int cnt = 1;
        for(int i=1;i<n;i++){
            if(array[i] == candidate){
                cnt++;
            }else {
                cnt --;
                if(cnt == 0){
                    candidate = array[i];
                    cnt =1;
                }
            }
        }
        int realTime = 0;
        for(int i=0;i<n;i++) {
            if (array[i] == candidate) {
                realTime ++;
                if(realTime > n / 2){
                    return candidate;
                }
            }
        }
        return 0;
    }
}
