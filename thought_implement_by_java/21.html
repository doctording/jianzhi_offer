import java.util.ArrayList;
import java.util.Stack;

public class Solution {
     public boolean IsPopOrder(int [] pushA,int [] popA) {
        int n = pushA.length;
        Stack<Integer> sta = new Stack<>();
        // 栈中目前存在的元素
        sta.push(pushA[0]);
        // 下一个要入栈的元素
        int nextIndex = 1;
        for(int i = 0;i < n; i++){
            if(sta.isEmpty()){
                sta.push(pushA[nextIndex]);
                nextIndex++;
                if(nextIndex >= n){
                    break;
                }
            }
            // 栈顶元素
            int staPeek = sta.peek();
            // 出栈的元素
            int popVal = popA[i];
            if(popVal > staPeek){
                if(nextIndex >= n){
                    return false;
                }
                while(pushA[nextIndex] <= popVal){
                    sta.push(pushA[nextIndex]);
                    nextIndex++;
                    if(nextIndex >= n){
                        break;
                    }
                }
                sta.pop();
            }else if(popVal == staPeek){
                sta.pop();
            }else {
                // 要pop的的元素小于栈顶元素，不符合
                return false;
            }
        }
        if(sta.empty()){
            return true;
        }
        return false;
    }
}
