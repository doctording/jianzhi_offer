注意题目描述

输入一个递增排序的数组和一个数字S，在数组中查找两个数，使得他们的和正好是S，如果有多对数字的和等于S，输出两个数的乘积最小的。

二分法

时间复杂度nlogn

import java.util.ArrayList;
public class Solution {
      int bsearch(int[] array, int left, int right, int val){
        if(val < array[left] || val > array[right]){
            return -1;
        }
        while(left <= right){
            int mid = (right - left) / 2 + left;
            if(array[mid] == val){
                return mid;
            }else if(array[mid] < val){
                left = mid + 1;
            }else {
                right = mid - 1;
            }
        }
        return -1;
    }

    public ArrayList<Integer> FindNumbersWithSum(int [] array,int sum) {
        ArrayList<Integer> rs = new ArrayList<>(2);
        int val1 = 0;
        int val2 = 0;
        int minSum = Integer.MAX_VALUE;
        boolean flag = false;
        int n = array.length;
        for(int i=0;i<n-1;i++){
            int vali = array[i];
            int pos2 = bsearch(array, i + 1, n - 1, sum - vali);
            if(pos2 != -1){
                if(vali * (sum - vali) < minSum){
                    flag = true;
                    val1 = vali < (sum - vali) ? vali : (sum - vali);
                    val2 = sum - val1;
                    minSum = vali * (sum - vali);
                }
            }
        }
        if(flag) {
            rs.add(val1);
            rs.add(val2);
        }
        return rs;
    }
}

双指针

import java.util.ArrayList;
public class Solution {
     public ArrayList<Integer> FindNumbersWithSum(int [] array,int sum) {
        ArrayList<Integer> rs = new ArrayList<>(2);
        int val1 = 0;
        int val2 = 0;
        int minSum = Integer.MAX_VALUE;
        boolean flag = false;
        int n = array.length;

        int iLeft = 0;
        int iRight = n- 1;

        while(iLeft < iRight){
            if(array[iLeft] + array[iRight] == sum){
                if(array[iLeft] * array[iRight] < minSum){
                    flag = true;
                    minSum = array[iLeft] * array[iRight];
                    val1 = array[iLeft];
                    val2 = array[iRight];
                }
                iLeft++;
                iRight--;
            }else if(array[iLeft] + array[iRight] < sum){
                iLeft ++;
            }else {
                iRight--;
            }
        }
        if(flag) {
            rs.add(val1);
            rs.add(val2);
        }
        return rs;
    }
}
