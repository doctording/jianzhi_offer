  1. 不考虑顺序,类似快排思路；前后两个游标，不断的查找交换

时间复杂度O(n)

 public void reOrderArray(int [] array) {
    int n = array.length;
    int i = 0;
    int j = n - 1;
    while(i < j){
        // 从左往右找偶数
        while( i < j && array[i] % 2 == 1){
            i++;
        }
        while( j > i && array[j] % 2 == 0){
            j--;
        }
        if(i < j){
            int tmp = array[i];
            array[i] = array[j];
            array[j] = tmp;
        }
    }
}
  1. 考虑顺序一致

插入排序，移动元素
时间复杂度O(n^2), 空间复杂度O(1)

public class Solution {
    public void reOrderArray(int [] array) {
        int n = array.length;
        // 已经有的奇数个数
        int k = 0;
        for(int i =0;i<n;i++){
            // 是奇数
            if(array[i] % 2 == 1){
                int val = array[i];
                int j = i;
                // 后移元素
                while(j > k){
                    array[j] = array[j-1];
                    j--;
                }
                array[k] = val;
                k++;
            }
        }
    }

}
  1. 归并排序
public class Solution {
    /**
     * 可以理解为，两个有序数组的合并排序
     * 思路：维持两个数组的游标，不断的比较
     * 当前比较规则是： 奇数在前，偶数后面
     */
    void merge2(int arr[], int sta, int mid, int end){
        if(sta >= end){
            return;
        }
        // sta mid -1
        // mid end
        int[] arrTmp = new int[end - sta + 1];
        // p1, p2 游标
        int p1 = sta;
        int p2 = mid + 1;
        int k = 0;
        while(p1 <= mid && p2 <= end){
//            if(arr[p1] <= arr[p2]) {
//                arrTmp[k++] = arr[p1++];
//            }else {
//                arrTmp[k++] = arr[p2++];
//            }
            if(arr[p1] % 2 == 1) {
                arrTmp[k++] = arr[p1++];
            }else {
                if(arr[p2] % 2 == 1) {
                    arrTmp[k++] = arr[p2++];
                }else{
                    // 都是偶数，p1优先；因为p1本来就在p2的前面，为了不改变顺序
                    arrTmp[k++] = arr[p1++];
                }
            }
        }
        while(p1 <= mid) {
            // 如果第一个序列未检测完，直接将后面所有元素加到合并的序列中
            arrTmp[k++] = arr[p1++];
        }
        while(p2 <= end){
            // 同上
            arrTmp[k++] = arr[p2++];
        }
        // 复制回去
        for (int i = sta; i <= end; i++) {
            arr[i] = arrTmp[i - sta];
        }
    }

    void mergeSort(int arr[], int sta, int end){
        if(sta >= end){
            return;
        }
        int mid = (end - sta) / 2 + sta;
        mergeSort(arr, sta, mid);
        mergeSort(arr, mid + 1, end);
        // 左右处理完，然后归并
        merge2(arr, sta, mid, end);
    }

    public void reOrderArray(int [] array) {
        int len = array.length;
        mergeSort(array, 0 ,len - 1);
    }
}
