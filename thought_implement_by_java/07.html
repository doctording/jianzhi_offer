快速幂求解

用二维数组表示矩阵(线性代数基础)

{
    {1, 1},
    {1, 0}
}

// 矩阵相乘

1 2 3       1 2
4 5 6  *    3 4
            5 6

public class Solution {

    // matrix = [[1,1],
    //      [1,0]]
    // [fn fn-1] = [f1 f0] * matrix^(n-1)
    long[][] multiMatrix(long[][] m1, long[][] m2){
        long[][] m = {
                {0, 0},
                {0, 0}
        };
        for(int i = 0;i < 2;i++){
            for(int j=0;j<2;j++){
                long tmp = 0;
                for(int k=0;k<2;k++){
                    tmp += m1[i][k] * m2[k][j];
                }
                m[i][j] = tmp;
            }
        }
        return m;
    }

    public int Fibonacci(int n) {
        if(n == 0){
            return 0;
        }
        if(n == 1 || n == 2){
            return 1;
        }
        // 全1矩阵
        long[][] rs = {
                {1, 0},
                {0, 1}
        };
        long[][] m = {
                {1, 1},
                {1, 0}
        };
        long tmpN = n - 2;
        while(tmpN != 0){
            long bit1 = tmpN & 1;
            if(bit1 == 1){
                rs = multiMatrix(rs, m);
            }
            m = multiMatrix(m, m);
            tmpN = tmpN >> 1;
        }
        int ans = 0;
        for(int i=0;i<2;i++){
            ans += rs[i][0];
        }
        return ans;
    }
}
