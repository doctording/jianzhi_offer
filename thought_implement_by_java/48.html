48 不用加减乘除做加法

public class Solution {
    public int Add(int num1,int num2) {
        int num = num1 ^ num2;
        int carry = (num1 & num2) << 1;
        while(carry != 0){
            int tmp1 = num ^ carry;
            int tmp2 = (num & carry) << 1;
            num = tmp1;
            carry = tmp2;
        }
        return num;
    }
}
