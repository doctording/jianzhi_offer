双端队列，滑动窗口

import java.util.*;

public class Solution {
    Deque<Integer> deque = new LinkedList<>();

    public ArrayList<Integer> maxInWindows(int [] num, int size) {
        if(size <= 0){
            return new ArrayList<>();
        }
        int n = num.length;
        if(size > n){
            return new ArrayList<>();
        }
        ArrayList<Integer> ans = new ArrayList<>();
        if(size == 1){
            for(int i=0;i<n;i++) {
                ans.add(num[i]);
            }
            return ans;
        }
        for(int i=0;i<n;i++){
            if(i == 0){
                deque.push(0);
            }else{
                int val = num[i];
                // 后面小的，删除掉
                while (!deque.isEmpty()){
                    int lastIndex = deque.getLast();
                    if(num[lastIndex] < val){
                        deque.pollLast();
                    }else {
                        break;
                    }
                }
                // 前面超过窗口了，也删除掉
                while (!deque.isEmpty()) {
                    int index = deque.getFirst();
                    if(i - index >= size){
                        deque.pollFirst();
                    }else{
                        break;
                    }
                }
                deque.addLast(i);
                if(i >= size - 1){
                    ans.add(num[deque.peekFirst()]);
                }
            }
        }
        return ans;
    }
}

push-pop-max栈，用两个这样的栈，实现一个队列

import java.util.Stack;
import java.util.ArrayList;

public class Solution {
     class MaxStack{
        Stack<Integer> stack = new Stack<>();
        Stack<Integer> maxStack = new Stack<>();

        public void push(int node) {
            if(stack.isEmpty()){
                stack.push(node);
                maxStack.push(node);
            }else {
                int lastMinVal = maxStack.peek();
                if(node >= lastMinVal){
                    stack.push(node);
                    maxStack.push(node);
                }else {
                    stack.push(node);
                }
            }
        }

        public int pop() {
            if(stack.empty()){
                return 0;
            }
            int topVal = stack.pop();
            int maxVal = maxStack.peek();
            if(topVal == maxVal){
                maxStack.pop();
            }
            return topVal;
        }

        public int top() {
            return stack.peek();
        }

        public int max() {
            return maxStack.peek();
        }

        public boolean empty() {
            return stack.empty();
        }

    }


    class MaxQueue{
        MaxStack stack1 = new MaxStack();
        MaxStack stack2 = new MaxStack();

        public void push(int node) {
            stack1.push(node);
        }

        public int pop() {
            while(!stack1.empty()){
                int tmp = stack1.pop();
                stack2.push(tmp);
            }

            int popVal = stack2.pop();

            while(!stack2.empty()){
                int tmp = stack2.pop();
                stack1.push(tmp);
            }

            return popVal;
        }

        public int max() {
            return stack1.max();
        }
    }

    public ArrayList<Integer> maxInWindows(int [] num, int size) {
        if(size <= 0){
            return new ArrayList<>();
        }
        MaxQueue maxQueue = new MaxQueue();
        int n = num.length;
        if(size > n){
            return new ArrayList<>();
        }
        for(int i=0;i<size-1;i++){
            maxQueue.push(num[i]);
        }
        ArrayList<Integer> ans = new ArrayList<>(n-size);
        for(int i=size-1;i<n;i++){
            maxQueue.push(num[i]);
            ans.add(maxQueue.max());
            maxQueue.pop();
        }
        return ans;
    }
}
