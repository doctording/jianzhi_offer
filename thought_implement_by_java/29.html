快排思路

import java.util.ArrayList;

public class Solution {
     int partition(int[] input, int left, int right){
        if(left > right){
            return -1;
        }
        int pivot = input[left];
        int iLeft = left;
        int iRight = right;
        while(iLeft < iRight){
            // 从右往左
            while (iLeft < iRight && input[iRight] >= pivot){
                --iRight;
            }
            input[iLeft] = input[iRight];
            // 从左往右
            while (iLeft < iRight && input[iLeft] < pivot){
                ++ iLeft;
            }
            input[iRight] = input[iLeft];

        }
        input[iLeft] = pivot;
        return iLeft;
    }

    ArrayList<Integer> reverse(int [] input, int left, int right, int k) {
        ArrayList<Integer> ans = new ArrayList<>(k);
        int p1 = partition(input, left, right);
        if(p1 - left == k){
            for(int i = left; i < p1; i ++){
                ans.add(input[i]);
            }
            return ans;
        }
        if(p1 - left + 1 == k){
            for(int i = left; i <= p1; i++){
                ans.add(input[i]);
            }
            return ans;
        }
        if(p1 - left + 1 < k){
            for(int i = left; i <= p1; i++){
                ans.add(input[i]);
            }
            int leftCnt = k - (p1-left+1);
            ans.addAll(reverse(input,p1 +1 ,right, leftCnt));
            return ans;
        }
        return reverse(input, left, p1-1, k);
    }

    public ArrayList<Integer> GetLeastNumbers_Solution(int [] input, int k) {
        int n = input.length;
        ArrayList<Integer> ans = new ArrayList<>(k);
        if(n == k){
           for(int i=0;i<n;i++){
               ans.add(input[i]);
           }
        }else if(n > k){
            int left = 0;
            int right = n - 1;
            ans = reverse(input, left, right, k);
        }
        return ans;
    }
}

堆(最大堆)

import java.util.ArrayList;

public class Solution {
    void adjustAPos(int[] heap, int n, int pos){
        int k = pos;
        while(k <= n/2){
            int l_k = 2 * k;
            int r_k = 2 * k + 1;
            if(r_k > n) {
                if(heap[l_k] > heap[k]) {
                    int tmp = heap[k];
                    heap[k] = heap[l_k];
                    heap[l_k] = tmp;
                }
                return;
            }
            if(heap[l_k] <= heap[k] && heap[r_k] <= heap[k]){
                return;
            }else if(heap[l_k] > heap[k] && heap[r_k] <= heap[k]){
                int tmp = heap[k];
                heap[k] = heap[l_k];
                heap[l_k] = tmp;
                k = l_k;
            }else if(heap[l_k] <= heap[k] && heap[r_k] > heap[k]){
                int tmp = heap[k];
                heap[k] = heap[r_k];
                heap[r_k] = tmp;
                k = r_k;
            }else {
                // 左边更大
                if(heap[l_k] >= heap[r_k]) {
                    int tmp = heap[k];
                    heap[k] = heap[l_k];
                    heap[l_k] = tmp;
                    k = l_k;
                }else { // 右边更大
                    int tmp = heap[k];
                    heap[k] = heap[r_k];
                    heap[r_k] = tmp;
                    k = r_k;
                }
            }

        }
    }

    void adjustAllPos(int[] heap, int n){
        for(int i=n/2;i>0;i--){
            adjustAPos(heap, n, i);
        }
    }

    public ArrayList<Integer> GetLeastNumbers_Solution(int [] input, int k) {
        int n = input.length;
        ArrayList<Integer> ans = new ArrayList<>(k);
        if(n == k){
           for(int i=0;i<n;i++){
               ans.add(input[i]);
           }
        }else if(n > k && k > 0){
            int[] maxHeap = new int[k + 1];
            for(int i=0;i<k;i++){
                maxHeap[i + 1] = input[i];
            }
            adjustAllPos(maxHeap, k);
            for(int i=k;i<n;i++){
                int heapMaxVal = maxHeap[1];
                if(input[i] >= heapMaxVal){
                    continue;
                }

                maxHeap[1] = input[i];
                adjustAllPos(maxHeap, k);
            }

            for(int i=1;i<=k;i++){
                ans.add(maxHeap[i]);
            }
        }
        return ans;
    }
}
