# -*- coding:utf-8 -*-
class Stack:
    def __init__(self) :
        self.items = []

    def empty(self) :
        if len(self.items) == 0 :
            return True 
        return False 

    def size(self) :
        return len(self.items)

    def push(self, data) :
        self.items.append(data)

    def pop(self) :
        self.items.pop(-1)

    def top(self) :
        return self.items[-1]

class Solution:
    def __init__(self) :
        self.sta1 = Stack() # 数据始终用sta1存储， sta2辅助pop
        self.sta2 = Stack()

    def push(self, node):
        # write code here
        self.sta1.push(node)

    def pop(self):
        # return xx
        if self.sta1.empty() :
            return None

        # 1：sta1 转储到 sta2
        while self.sta1.empty() == False :
            node = self.sta1.top()
            self.sta1.pop()

            self.sta2.push(node)

        # 2: 出队列的数据  
        val = self.sta2.top()
        self.sta2.pop()

        # 3：sta2 转储到 sta1
        while self.sta2.empty() == False :
            node2 = self.sta2.top()
            self.sta2.pop()

            self.sta1.push(node2)

        return val
