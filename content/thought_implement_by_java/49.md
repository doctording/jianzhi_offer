public class Solution {
    boolean isdigit(char c){
        if(c >= '0' && c <= '9'){
            return true;
        }
        return false;
    }

    public int StrToInt(String str) {
        if(null == str || str.isEmpty()){
            return 0;
        }

        str = str.trim();
        int sign = 1;
        char[] chars = str.toCharArray();
        if(chars[0] == '+'){
            str = str.substring(1);
        }else if(chars[0] == '-'){
            str = str.substring(1);
            sign = -1;
        }

        chars = str.toCharArray();
        int len = chars.length;
        for(int i=0;i<len;i++){
            if(!isdigit(chars[i])){
                return 0;
            }
        }

        int maxInt = Integer.MAX_VALUE;
        int minInt = Integer.MIN_VALUE;

        int num = 0;
        for(int i=0;i<len;i++){
            int nowC = chars[i] - '0'; // 注意变成数字
            if(sign == 1){
                // 正数处理
                if(num < maxInt / 10){
                    num = num * 10 + nowC;
                }else if(num == maxInt / 10){
                    if(nowC > maxInt % 10){
                        return maxInt;
                    }else {
                        num = num * 10 + nowC;
                    }
                }else{
                    return maxInt;
                }
            }else{
                // 负数的处理
                if(num < maxInt / 10){
                    num = num * 10 + nowC;
                }else if(num == maxInt / 10){
                    if(nowC >= (maxInt % 10 + 1) ){
                        return minInt;
                    }else {
                        num = num * 10 + nowC;
                    }
                }else{
                    return minInt;
                }
            }
        }
        return sign * num;
    }
}
