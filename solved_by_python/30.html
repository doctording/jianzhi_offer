动态递推

# -*- coding:utf-8 -*-
class Solution:
    def FindGreatestSumOfSubArray(self, array):
        # write code here
        n = len(array)
        li = [0]
        li = li * n

        li[0] = array[0]
        maxVal = li[0]

        i = 1
        while i < n :
            if li[i-1] + array[i] > array[i] :
                li[i] = li[i - 1] + array[i]
            else :
                li[i] = array[i]

            if li[i] > maxVal :
                maxVal = li[i]
            i += 1
        return maxVal
