解法2

# -*- coding:utf-8 -*-
class Solution:
    # s 源字符串
    def replaceSpace(self, s):
        # write code here
        n = len(s)
        s2 = ""
        i1 = n - 1
        while i1 >= 0 :
            if s[i1] == ' ':
                s2 = "%20" + s2
            else:
                s2 = s[i1] + s2
            i1 = i1 - 1
        return s2
