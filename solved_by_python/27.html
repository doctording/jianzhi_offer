DFS求解

# -*- coding:utf-8 -*-
class Solution:
    def Permutation(self, ss):
        # write code here
        ans = []
        n = len(ss)
        if n == 0 :
            return ans

        def dfs(ss, index, n) :

            if index == n :
                ans.append(ss)
                return None

            i = index
            while i < n :
                if i > index and ss[i] == ss[index] : # 去重复
                    i += 1
                    continue

                s2_save = ss

                s2 = list(ss)
                tmp = s2[i]
                s2[i] = s2[index]
                s2[index] = tmp
                ss = ''.join(s2)

                dfs(ss, index + 1, n) # 递归

                # 回溯
                ss = s2_save

                i += 1
        dfs(ss, 0, n) # call
        ans.sort() # list 排序
        return ans
