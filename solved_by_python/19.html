# -*- coding:utf-8 -*-
class Solution:

    def myPrint(self, matrix, x1, y1, x2, y2) :
        li = []
        if x1 > x2 or y1 > y2 :
            return li

        if x1 == x2 :
            li = matrix[x1][y1:y2+1]
        elif y1 == y2 :
            i = x1
            while i <= x2 :
                li.append(matrix[i][y1])
                i += 1
        else :
            li = matrix[x1][y1:y2+1]

            i = x1 + 1
            while i < x2 :
                li.append(matrix[i][y2])
                i += 1

            i = y2
            while i >= y1 :
                li.append(matrix[x2][i])
                i -= 1

            i = x2 - 1
            while i > x1 :
                li.append(matrix[i][y1])
                i -= 1
        return li

    # matrix类型为二维列表，需要返回列表
    def printMatrix(self, matrix):
        # write code here
        m = len(matrix)
        if m == 0 :
            pass
        n = len(matrix[0])
        if n == 0 :
            pass

        ans = []
        x1 = 0
        y1 = 0
        x2 = m - 1
        y2 = n - 1
        cnt = 0
        while cnt < m * n :
            ans = ans + self.myPrint(matrix, x1, y1, x2, y2)
            if x1 == x2 :
                cnt += y2 - y1 + 1
            elif y1 == y2 :
                cnt += x2 - x1 + 1
            else:
                cnt += 2 * (y2 - y1 + 1) + 2 * (x2 - x1 - 1)
            x1 += 1
            y1 += 1
            x2 -= 1
            y2 -= 1

        return ans
