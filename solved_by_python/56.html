解法1

  • 链表一次变量，需要处理特殊情况
# -*- coding:utf-8 -*-
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None
class Solution:
    def deleteDuplication(self, pHead):
        # write code here
        if pHead == None or pHead.next == None :
            return pHead

        # 全部相同
        node = pHead.next
        while node != None and node.val == pHead.val :
            node = node.next
        if node == None :
            return None

        # 处理头部, 补充一个新节点
        pTmp = ListNode(-1)
        pTmp.next = pHead
        pHead= pTmp

        p2 = pHead
        p2Tail = pHead.next
        lastVal = p2Tail.val

        p_pre = pHead
        p = pHead.next.next
        p2Tail.next = None

        while p != None :
            pNext = p.next
            if p.val != lastVal :
                p_pre = p2Tail
                p2Tail.next = p
                p2Tail = p
                p2Tail.next= None
                lastVal = p.val
            else :
                p2Tail = p_pre
                p2Tail.next = None
                p_pre.next = None
                lastVal = p_pre.val
            p = pNext

        return p2.next

借助 map

  • 先统一值的情况，然后遍历一次构造链表
# -*- coding:utf-8 -*-
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None
class Solution:
    def deleteDuplication(self, pHead):
        # write code here
        if pHead == None or pHead.next == None :
            return pHead

        mp = {}
        p = pHead
        while p != None :
            if mp.has_key(p.val) :
                mp[p.val] += 1
            else :
                mp[p.val] = 1
            p = p.next

        p2 = ListNode(-1)
        p2Tail = p2

        p = pHead
        while p != None :
            pNext = p.next
            if mp[p.val] == 1 :
                p2Tail.next = p
                p2Tail = p
                p2Tail.next = None
            p = pNext
        return p2.next
