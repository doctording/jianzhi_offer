二叉树层次遍历

# -*- coding:utf-8 -*-
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None
class Solution:
    # 返回从上到下每个节点值列表，例：[1,2,3]
    def PrintFromTopToBottom(self, root):
        if root == None :
            return []

        que = []
        que_size = 1
        que.append(root)
        ans = []

        while que_size != 0 :
            node = que[0]
            que_size -= 1
            que.pop(0)

            ans.append(node.val)

            if node.left != None :
                que.append(node.left)
                que_size += 1

            if node.right != None :
                que.append(node.right)
                que_size += 1
        return ans
