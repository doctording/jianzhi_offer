2、3、5因子

# -*- coding:utf-8 -*-
class Solution:
    def GetUglyNumber_Solution(self, index):
        if index == 0 :
            return 0
        li = [1]

        pos2 = 0
        pos3 = 0
        pos5 = 0

        i = 1
        while i < index :
            v2 = li[pos2] * 2
            v3 = li[pos3] * 3
            v5 = li[pos5] * 5

            minVal = min(v2, v3)
            minVal = min(minVal, v5)

            if v2 <= minVal :
                pos2 += 1

            if v3 <= minVal :
                pos3 += 1

            if v5 <= minVal :
                pos5 += 1

            li.append(minVal)
            i += 1

        return li[index - 1]
