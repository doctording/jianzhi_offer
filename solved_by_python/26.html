解法1

递归求出中序遍历序列，然后处理左右节点

# -*- coding:utf-8 -*-
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None
class Solution:
    def Convert(self, pRootOfTree):

        if pRootOfTree == None :
            return None

        ans = []

        def inOrder(root) :
            if root != None :
                inOrder(root.left)
                ans.append(root)
                inOrder(root.right)
        inOrder(pRootOfTree) # call inOrder

        head = TreeNode(-1)
        tail = head
        pre = head

        for node in ans :
            tail.right = node
            tail = node
            tail.left = pre
            pre = pre.right

        h2 = head.right
        h2.left = None
        return h2

解法2

利用栈，非递归遍历二叉树

# -*- coding:utf-8 -*-
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None
class Solution:
    def Convert(self, pRootOfTree):

        if pRootOfTree == None :
            return None

        sta = []
        sta.append(pRootOfTree)
        sta_size = 1

        pLeft = pRootOfTree.left

        h2 = TreeNode(-1)
        t2 = h2
        pre = h2

        while pLeft != None or sta_size != 0 :
            while pLeft != None :
                sta.append(pLeft)
                sta_size += 1

                pLeft = pLeft.left

            topNode = sta[sta_size - 1]
            sta.pop(-1)
            sta_size -= 1

            t2.right = topNode
            topNode.left = t2

            t2 = topNode

            if topNode.right != None :
                pLeft = topNode.right

        ans = h2.right
        ans.left = None
        return ans
