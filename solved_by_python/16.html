# -*- coding:utf-8 -*-
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None
class Solution:
    # 返回合并后列表
    def Merge(self, pHead1, pHead2):
        # write code here
        if pHead1 == None :
            return pHead2
        if pHead2 == None :
            return pHead1

        h2 = ListNode(-1)
        t2 = h2

        p1 = pHead1
        p2 = pHead2

        while p1 != None and p2 != None :
            p1Next = p1.next
            p2Next = p2.next
            if p1.val < p2.val :
                p1.next = None
                t2.next = p1
                t2 = p1

                p1 = p1Next
            elif p1.val > p2.val :
                p2.next = None
                t2.next = p2
                t2 = p2

                p2 = p2Next
            else :
                p1.next = None
                t2.next = p1
                t2 = p1

                p2.next = None
                t2.next = p2
                t2 = p2

                p1 = p1Next
                p2 = p2Next

        if p1 != None :
            while p1 != None :
                p1Next = p1.next

                p1.next = None
                t2.next = p1
                t2 = p1

                p1 = p1Next

        if p2 != None :
             while p2 != None :
                p2Next = p2.next

                p2.next = None
                t2.next = p2
                t2 = p2

                p2 = p2Next

        return h2.next
