解法1(AC)

主要分2步：1，深拷贝单链表包括next；2，处理random

  • 不改变原链表的结构

  • 利用了Map的数据结构

# -*- coding:utf-8 -*-
# class RandomListNode:
#     def __init__(self, x):
#         self.label = x
#         self.next = None
#         self.random = None
class Solution:
    # 返回 RandomListNode
    def Clone(self, pHead):
        # write code here
        pHead2 = RandomListNode(-1)
        pTail2 = pHead2

        # 1 构造
        p1 = pHead
        while p1 != None :
            node = RandomListNode(p1.label)
            pTail2.next = node
            pTail2 = node

            p1 = p1.next

        # 2 map
        mp = {}
        p1 = pHead
        p2 = pHead2.next
        while p1 != None :
            mp[p1] = p2
            p1 = p1.next
            p2 = p2.next

        # 3 random
        p1 = pHead
        p2 = pHead2.next
        while p1 != None :
            if p1.random == None :
                p2.random = None
            else :
                p2.random = mp[p1.random]
            p1 = p1.next
            p2 = p2.next

        return pHead2.next

方法2(WA)

  • 改变了原链表，不通过；不过思路值得参考
# -*- coding:utf-8 -*-
# class RandomListNode:
#     def __init__(self, x):
#         self.label = x
#         self.next = None
#         self.random = None
class Solution:
    # 返回 RandomListNode
    def Clone(self, pHead):

        # 1 原节点的next 指向 新节点，新节点的next 指向 原节点的 next
        p = pHead

        while p != None :
            pNext = p.next

            node = RandomListNode(p.label)
            node.next = pNext
            p.next = node

            p = pNext

        # 2 处理random
        p = pHead
        while p != None :
            pNew = p.next
            pOldNext = pNew.next

            pRandom = p.random
            if pRandom == None :
                pNew.random = None
            else :
                pNew.random = pRandom.next

            p = pOldNext

        # 3 处理 next
        p = pHead.next
        while p != None :
            pOld = p.next
            if pOld == None :
                break

            pNext = pOld.next

            p.next = pNext

            p = pNext

        pHead2 = pHead.next
        return pHead2
