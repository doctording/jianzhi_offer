按位模拟，位运算(C++)

class Solution {
public:
    int Add(int num1, int num2)
    {
        while(num2 != 0 ){
            int tmp = num1 ^ num2 ;// 两数相加，不进位
            num2  = (num1 & num2) << 1 ;// 进位
            num1 = tmp;
        }
        return num1;
    }
};
