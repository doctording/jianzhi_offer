类似斐波那契，O(n)的方法不行，必须采用快速幂的方法

# -*- coding:utf-8 -*-
def multi(m1, m2) :
    m = [[0,0],[0,0]]
    for i in range(0,2) :
        for j in range(0,2) :
            for k in range(0, 2) :
                m[i][j] += m1[i][k] * m2[k][j]
    return m

def multiN(m, n) :
    rs = [[1, 0],[0, 1]]
    while n > 0 :
        if n & 1 == 1 :
            rs = multi(rs, m)
        m = multi(m, m)
        n = n >> 1
    return rs

class Solution:
    def jumpFloor(self, number):
        # write code here
        if number == 0 :
            return 0
        if number == 1 :
            return 1
        if number == 2 :
            return 2

        m = [[1,1],[1,0]]
        # [fn fn-1] = [f2 f1] * matrix^(n-2)
        rs = multiN(m, number-2)
        ans = rs[0][0] * 2 + rs[1][0]
        return ans
