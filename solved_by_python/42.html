二分查找求解

# -*- coding:utf-8 -*-
class Solution:
    def FindNumbersWithSum(self, array, tsum):
        # write code here
        def bsearch(array, left, right, val) :
            if left > right :
                return -1
            while left <= right :
                mid = (left + right) / 2
                if array[mid] < val :
                    left = mid + 1
                elif array[mid] > val :
                    right = mid - 1
                else :
                    return mid
            return -1

        maxTmp = 0x7fffffff
        ans = []
        n = len(array)
        i = 0
        while i < n-1 :
            val1 = array[i]
            val2 = tsum - val1
            pos = bsearch(array, i + 1, n -1 ,val2)
            if pos != -1 :
                if val1 * val2 < maxTmp :
                    maxTmp = val1 * val2
                    ans.append(val1)
                    ans.append(val2)
            i += 1

        return ans
