# -*- coding:utf-8 -*-
class Solution:
    # s字符串
    def isNumeric(self, s):
        # write code here
        n = len(s)
        i = 0
        dot_cnt = 0
        e_cnt = 0
        e_after = False
        e_pos = -1
        while i < n :
            if s[i] == '.' :
                if e_cnt > 0 : # e之后的数不能有小数点
                    return False
                dot_cnt += 1
                if dot_cnt > 1 :
                    return False # 2个或以上的小数点错误

            elif s[i] == 'e' or s[i] == 'E' :
                e_pos = i
                e_cnt += 1
                if e_cnt > 1 : # e只能有一个
                    return False
            elif s[i] == '-' or s[i] == '+' :
                if e_pos != -1 and i == e_pos + 1 :
                    pass
                elif e_pos != -1 and i != e_pos + 1 :
                    return False
                elif e_pos == -1 and i == 0 :
                    pass
                elif e_pos == -1 and i > 0 :
                    return False
                else :
                    pass
            elif s[i] >= '0' and s[i] <= '9' :
                if e_cnt > 0 :
                    e_after = True
            else :
                return False # 出现其它字符，错误
            i += 1

        if e_cnt > 0 and e_after == False : # 存在e，则e之后必须有数字
            return False
        return True
