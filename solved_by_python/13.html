# -*- coding:utf-8 -*-
class Solution:
    def reOrderArray(self, array):
        # write code here
        l1 = []
        l2 = []
        for n in array:
            if n & 1 == 1:
                l1.append(n)
            else:
                l2.append(n)
        return l1 + l2
