序列化

  • 定义
You may serialize the following tree:

    1
   / \
  2   3
     / \
    4   5

as "[1,2,3,null,null,4,5]"
  • list下标对应fa，left，right
       0
     /   \
   1      2
  / \    /  \
 3   4  5    6

i
left = 2 * (i + 1) - 1
right = 2 * (i + 1)

fa
left / 2
right / 2 - 1
  • BFS 反序列化问题
     5
    /
   4
  /
 3
  \
   2

序列化结果： "['5', '4', 'null', '3', 'null', 'null', '2']"

i  le ri
0  1  2
1  3  4
2
3  5  6

初始 i = 0 ， le = 1;
接着 i += 1, le += 2 , 即i = 1时，le = 3
再接着 i += 1, le += 2, 即 i = 2 时 ,le = 5
但是 i位置节点为None, 所以接着 i += 1，没有 le += 2
即 i = 3 时， 仍有 le = 5，（符合预期）

class Solution:
    def Serialize(self, root):
        # write code here
        s = "["
        if root == None:
            s += "]"
            return s

        que = []
        que.append(root)
        que_size = 1

        while que_size != 0:
            que2 = []
            que_size2 = 0
            i = 0
            while i < que_size:
                node = que[i]
                if node != None:
                    s += str(node.val) + ","

                    if node.left != None :
                        que2.append(node.left)
                    else :
                        que2.append(None)

                    if node.right != None:
                        que2.append(node.right)
                    else:
                        que2.append(None)

                    que_size2 += 2
                else:
                    s += "null,"

                i += 1
            que_size =  que_size2
            que = que2

        # 去除无效的 null和,
        n = len(s)
        i = n - 1
        while i >= 0 :
            if s[i]>='0' and s[i] <= '9' :
                break
            i -= 1
        s = s[0:i + 1]
        s += "]"
        return s

    def Deserialize(self, s):
        if s == "[]" :
            return None

        n = len(s)
        s = s[1:n-1]

        li = s.split(",") # 分割字符串

        print li
        # 转换成 TreeNode
        treeli = []
        for s in li :
            if s == "null" :
                treeli.append(None)
            else :
                val = int(s)
                node = TreeNode(val)
                treeli.append(node)

        treeli_size = len(treeli)

        index = 0
        le = 2 * (index + 1) - 1
        while index < treeli_size :
            if treeli[index] == None :
                index += 1
            else :
                if le < treeli_size :
                    treeli[index].left = treeli[le]
                if le + 1 < treeli_size :
                    treeli[index].right = treeli[le + 1]
                index += 1
                le += 2

        return treeli[0]
