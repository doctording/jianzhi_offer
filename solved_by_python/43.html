利用python str,list直接求解

# -*- coding:utf-8 -*-
class Solution:
    def LeftRotateString(self, s, n):
        # write code here
        li = list(s)

        l1 = li[0:n]
        l2 = li[n:]

        l = l2 + l1

        return ''.join(l)

三步反转

# -*- coding:utf-8 -*-
class Solution:
    def LeftRotateString(self, s, n):
        # write code here
        li = list(s)
        l1 = li[0:n]
        l2 = li[n:]

        l1.reverse()
        l2.reverse()

        l = l1 + l2
        l.reverse()

        return ''.join(l)
