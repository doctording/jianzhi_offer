01 二维数组中的查找
Table of Contents

解法1

从右上角开始直到左下角

public class Solution {
    public boolean Find(int target, int [][] array) {
        int m = array.length;
        if(m == 0){
            return false;
        }
        int n = array[0].length;
        if(n == 0){
            return false;
        }

        int i = 0;
        int j = n - 1;
        while(i < m && j >= 0) {
            if(array[i][j] > target){
                j --;
            }else if(array[i][j] < target){
                i ++;
            }else {
                return true;
            }
        }

        return false;
    }
}
