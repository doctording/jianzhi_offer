使用一个栈，不使用辅助栈

import java.util.Stack;

public class Solution {

    Stack<Integer> stack = new Stack<>();

    public void push(int node) {
        if(stack.isEmpty()){
            stack.push(0); // 当前元素与当前栈中最小元素的差值
            stack.push(node); // 当前最小值
        }else {
            int minVal = stack.pop();
            int diff = node - minVal;
            stack.push(diff);
            if(diff <= 0){ // 当前值node更小
                stack.push(node);
            }else {
                // 当前值node大，diff是正数 diff + minVal = lastVal
                stack.push(minVal);
            }
        }
    }

    public void pop() {
        int minVal = stack.pop();
        int diff = stack.pop();
        if(stack.isEmpty()){
            return;
        }
        if(diff <= 0){ // 最后一个元素是最小值， 紧接着的最小值是 minVal - diff
            stack.push(minVal - diff);
        }else {
            stack.push(minVal);
        }
    }

    public int top() {
        int minVal = stack.pop();
        int diff = stack.peek();
        int lastVal;
        if(diff <= 0){ // 最后一个元素就是最小值
            lastVal = minVal;
        }else {
            lastVal = minVal + diff;
        }
        stack.push(minVal);
        return lastVal;
    }

    public int min() {
        return stack.peek();
    }
}

使用一个普通栈和一个辅助栈

import java.util.Stack;

public class Solution {
    Stack<Integer> stack = new Stack<>();
    Stack<Integer> minStack = new Stack<>();

    public void push(int node) {
        if(stack.isEmpty()){
            stack.push(node);
            minStack.push(node);
        }else {
            int lastMinVal = minStack.peek();
            if(node <= lastMinVal){ // 当前值更小
                stack.push(node);
                minStack.push(node);
            }else {
                // 当前值node大, lastMinVal不变
                stack.push(node);
            }
        }
    }

    public void pop() {
        if(stack.empty()){
            return;
        }
        int topVal = stack.pop();
        int minVal = minStack.peek();
        if(topVal == minVal){
            minStack.pop();
        }
    }

    public int top() {
        return stack.peek();
    }

    public int min() {
        return minStack.peek();
    }
}
