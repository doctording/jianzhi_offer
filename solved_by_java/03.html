利用 Stack

import java.util.Stack;
/**
*    public class ListNode {
*        int val;
*        ListNode next = null;
*
*        ListNode(int val) {
*            this.val = val;
*        }
*    }
*
*/
import java.util.ArrayList;
public class Solution {
    public ArrayList<Integer> printListFromTailToHead(ListNode listNode) {
        ArrayList<Integer> rs = new ArrayList<>();

        if(listNode == null){
            return rs;
        }
        Stack<Integer> sta = new Stack<>();
        ListNode p = listNode;
        while(p != null){
            sta.push(p.val);
            p = p.next;
        }
        while(!sta.empty()){
            Integer val = sta.pop();
            rs.add(val);
        }
        return rs;
    }
}

递归

import java.util.Stack;
/**
*    public class ListNode {
*        int val;
*        ListNode next = null;
*
*        ListNode(int val) {
*            this.val = val;
*        }
*    }
*
*/
import java.util.ArrayList;
public class Solution {
   public ArrayList<Integer> rev(ListNode listNode){
        ArrayList<Integer> rs = new ArrayList<>();
        if(listNode == null){
            return rs;
        }else{
            rs = rev(listNode.next);
            rs.add(listNode.val);
            return rs;
        }
    }

    public ArrayList<Integer> printListFromTailToHead(ListNode listNode) {
        return rev(listNode);
    }
}
