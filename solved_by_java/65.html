深度遍历

设置访问标记，成功的全局判断标记；DFS深度搜索+回溯

public class Solution {
    int[] dx = {1,-1,0,0};
    int[] dy = {0,0,1,-1};

    boolean isOkPos(int rows, int cols, int x, int y){
        if(x < 0 || x >= rows || y < 0 || y >= cols){
            return false;
        }
        return true;
    }

    boolean flag = false;

    void dfs(boolean[] vis, char[] matrix, int rows, int cols, int nowPos, char[] str, int pos){
        if(flag){
            return;
        }
        if(pos == str.length - 1){
            flag = true;
            return;
        }
        int x = nowPos / cols;
        int y = nowPos % cols;
        for(int i=0;i<4;i++){
            int nx = x + dx[i];
            int ny = y + dy[i];
            if(isOkPos(rows, cols, nx, ny)){
                int nextPos = nx  * cols + ny;
                if(!vis[nextPos] && matrix[nextPos] == str[pos+1]){
                    vis[nextPos] = true;
                    dfs(vis,matrix,rows,cols, nextPos, str, pos+1);
                    if(flag){
                        return;
                    }
                    vis[nextPos] = false;
                }
            }
        }
    }

    public boolean hasPath(char[] matrix, int rows, int cols, char[] str)
    {
        for(int i=0;i<rows;i++){
            for (int j=0;j<cols;j++){
                int pos = i * cols + j;
                boolean[] vis = new boolean[rows * cols];
                if(matrix[pos] == str[0]){
                    vis[pos] = true;
                    dfs(vis,matrix, rows,cols, pos, str, 0);
                    if(flag){
                        return true;
                    }
                }
            }
        }
        return false;
    }
}
