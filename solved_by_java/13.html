 public void reOrderArray(int [] array) {
    int n = array.length;
    int i = 0;
    int j = n - 1;
    while(i < j){
        // 从左往右找偶数
        while( i < j && array[i] % 2 == 1){
            i++;
        }
        while( j > i && array[j] % 2 == 0){
            j--;
        }
        if(i < j){
            int tmp = array[i];
            array[i] = array[j];
            array[j] = tmp;
        }
    }
}

public class Solution {
    public void reOrderArray(int [] array) {
        int n = array.length;
        // 已经有的奇数个数
        int k = 0;
        for(int i =0;i<n;i++){
            // 是奇数
            if(array[i] % 2 == 1){
                int val = array[i];
                int j = i;
                // 后移元素
                while(j > k){
                    array[j] = array[j-1];
                    j--;
                }
                array[k] = val;
                k++;
            }
        }
    }

}
