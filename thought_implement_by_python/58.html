递归判断求解

# -*- coding:utf-8 -*-
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None
class Solution:

    def isSameTree(self, root1, root2) :
        if root1 == None and root2 == None :
            return True
        elif root1 == None and root2 != None :
            return False
        elif root1 != None and root2 == None :
            return False
        else :
            if root1.val != root2.val :
                return False
            b1 = self.isSameTree(root1.left, root2.right)
            b2 = self.isSameTree(root1.right , root2.left)
            return b1 and b2

    def isSymmetrical(self, pRoot):
        # write code here
        if pRoot == None :
            return True
        return self.isSameTree(pRoot.left, pRoot.right)

Java 求解

/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public boolean isSymmetric(TreeNode root) {
        if(root == null){
            return true;
        }
        return isSymRoot(root.left, root.right);
    }

    boolean isSymRoot(TreeNode t1, TreeNode t2) {
        if(t1 == null && t2 == null){
            return true;
        }else if(t1 == null && t2 != null){
            return false;
        }else if(t1 != null && t2 == null){
            return false;
        }else{
            if(t1.val != t2.val){
                return false;
            }
            return isSymRoot(t1.left, t2.right) && isSymRoot(t1.right, t2.left);
        }
    }
}
