堆求解 Top K 问题

# -*- coding:utf-8 -*-
def adjust(heap, n, pos):
    '''
    数组构成最大堆，调整
    :param heap: 
    :param n: 
    :param pos: 
    :return: 
    '''
    k = pos
    while k <= n / 2:
        l_k = 2 * k
        r_k = 2 * k + 1
        if r_k > n:
            if heap[l_k] > heap[k]:
                tmp = heap[k]
                heap[k] = heap[l_k]
                heap[l_k] = tmp
            return None

        if heap[k] >= heap[l_k] and heap[k] >= heap[r_k]:
            break
        elif heap[k] >= heap[l_k] and heap[k] < heap[r_k]:  # 右边大
            tmp = heap[k]
            heap[k] = heap[r_k]
            heap[r_k] = tmp
            k = r_k
        elif heap[k] < heap[l_k] and heap[k] >= heap[r_k]:  # 左边大
            tmp = heap[k]
            heap[k] = heap[l_k]
            heap[l_k] = tmp
            k = l_k
        else:
            if heap[l_k] >= heap[r_k]:  # 左边更大
                tmp = heap[k]
                heap[k] = heap[l_k]
                heap[l_k] = tmp
                k = l_k
            else:  # 右边更大
                tmp = heap[k]
                heap[k] = heap[r_k]
                heap[r_k] = tmp
                k = r_k

class Solution:
    def GetLeastNumbers_Solution(self, tinput, k):
        # write code here
        n = len(tinput)
        if k > n or k == 0 :
            return []

        heap = [-1]
        heap = heap + tinput[0:k]

        # 构造初始化大堆
        j = k / 2
        while j >= 1:
            adjust(heap, k, j)
            j -= 1

        #print heap

        i = k
        while i < n:
            if tinput[i] >= heap[1] :
                i += 1
                continue

            heap[1] = tinput[i]
            # 调整最小堆
            j = k / 2
            while j >= 1 :
                adjust(heap, k, j)
                j -= 1

            #print heap

            i += 1

        heap = heap[1:]
        heap.sort()
        return heap
