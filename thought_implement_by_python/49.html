字符串转int

注意符号和正负数范围

# -*- coding:utf-8 -*-
class Solution:
    def StrToInt(self, s):
        # write code here
        max_int = 0x7fffffff
        min_int = - max_int - 1

        def isDigit(c) :
            if c >= '0' and c <= '9' :
                return True
            return False

        s = s.strip() # 去除空白字符串
        slen = len(s)
        if slen == 0 :
            return 0

        flag = 1    # 判断正,负
        if s[0] == '-' :
            flag = -1
            s = s[1:]
        elif s[0] == '+' :
            s = s[1:]
        else :
            pass

        n = len(s)
        if n == 0 :
            return 0

        i = 0
        while i < n :
            if isDigit(s[i]) == False:
                return 0
            i += 1

        num = 0
        i = 0
        while i < n :
            nc = int(s[i])    # 当前数字
            if flag == 1 :    # 正数
                if num < max_int / 10 :
                    num = num * 10 + nc
                elif num == max_int / 10 :
                    if nc <= max_int % 10 :
                        num = num * 10 + nc
                    else :
                        num = max_int
                    return num #
                else :
                    return max_int #
            else :    # 负数
                if num < max_int / 10 :
                    num = num * 10 + nc
                elif num == max_int / 10 :
                    if nc <= max_int % 10 + 1:
                        num = num * 10 + nc
                        return - num #
                    else :
                        return min_int #
                else :
                    return min_int
            i += 1
        return flag * num
