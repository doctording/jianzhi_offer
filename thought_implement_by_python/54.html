解法 1

  • 时间复杂度

FirstAppearingOnce : O(n)
Insert: O(1)

  • 空间复杂度 O(n)
# -*- coding:utf-8 -*-
class Solution:
    def __init__(self) :
        self.li = []
        self.mp = {}
        self.li_cnt = 0

    # 返回对应char
    def FirstAppearingOnce(self):
        # write code here
        i = 0
        while i < self.li_cnt :
            if self.mp[self.li[i]] == 1 :
                return self.li[i]
            i += 1
        return '#'

    def Insert(self, char):
        # write code here
        self.li.append(char)
        self.li_cnt += 1
        if self.mp.has_key(char) :
            self.mp[char] += 1
        else :
            self.mp[char] = 1

解法 2

  • 时间复杂度

FirstAppearingOnce : O(1)
Insert: O(n)

  • 空间复杂度 O(n)
# -*- coding:utf-8 -*-
class Solution:
    def __init__(self) :
        self.li = []
        self.mp = {}
        self.li_cnt = 0

    # 返回对应char
    def FirstAppearingOnce(self):
        # write code here
        if self.li_cnt == 0:
            return '#'
        else:
            return self.li[0]

    def Insert(self, char):
        # write code here
        if self.mp.has_key(char) :
            if self.mp[char] == -1 : # 原来就出现多次
                pass
            else : # 原来只出现一次
                char_pos = self.mp[char]
                self.li.pop(char_pos)
                self.li_cnt -= 1
                self.mp[char] = -1 # 置为出现多次
                i = 0
                while i < self.li_cnt : # 更新位置
                    self.mp[self.li[i]] = i
                    i += 1
        else :
            self.li.append(char)
            self.mp[char] = self.li_cnt
            self.li_cnt += 1

解法 3

自定义链表结构

# -*- coding:utf-8 -*-
class Node :
    def __init__(self,data):
        self.val = data
        self.cnt = 1
        self.next = None
        self.pre = None

class ListNode:
    def __init__(self):
        self.root = Node(-1)
        self.tail = self.root
        self.size = 0

    def add(self, data):
        node = Node(data)

        p = self.root
        p = p.next
        while p != None :
            if p.val == data : # 原来存在这个元素，直接删除
                p_pre = p.pre
                p_pre.next = p.next
                if p.next == None:
                    self.tail = p_pre
                else:
                    p.next.pre = p_pre
                self.size -= 1
                return None
            p = p.next

        node.pre = self.tail
        self.tail.next = node
        self.tail = node
        self.size += 1

    def delete(self, data):
        # 直接删除元素
        p = self.root
        p = p.next
        while p != None:
            if p.val == data:
                p_pre = p.pre
                p_pre.next = p.next
                if p.next == None:
                    self.tail = p_pre
                else :
                    p.next.pre = p_pre
                self.size -= 1
                return None
            p = p.next

class Solution:
    def __init__(self) :
        self.li = ListNode()

    # 返回对应char
    def FirstAppearingOnce(self):
        if self.li.size == 0:
            return '#'
        else:
            return self.li.root.next.val

    def Insert(self, char):
        self.li.add(char)
