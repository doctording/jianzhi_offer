头插法反转

# -*- coding:utf-8 -*-
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None
class Solution:
    # 返回ListNode
    def ReverseList(self, pHead):
        # write code here
        if pHead == None or pHead.next == None :
            return pHead

        h2 = ListNode(-1)

        p = pHead
        while p != None:
            pNext = p.next

            p.next = h2.next
            h2.next = p

            p = pNext

        return h2.next
