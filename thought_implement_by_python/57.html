57 二叉树的下一个结点，中序

解法1

如下的二叉树，中序遍历： 7 2 5 1 8 9

    8
   / \
  7   9
   \
    5
   / \
  2   1

# -*- coding:utf-8 -*-
# class TreeLinkNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None
#         self.next = None
class Solution:
    def GetNext(self, pNode):
        if pNode == None :
            return None

        if pNode.right != None : # 右子树的最左节点
            pRoot = pNode.right
            while pRoot.left != None :
                pRoot = pRoot.left
            return pRoot

        while pNode.next != None: # 该节点不是该根节点
            if pNode == pNode.next.left : # 是父节点的左孩子，直接可以返回根
                return pNode.next
            pNode = pNode.next

        return None
