解法1

# -*- coding:utf-8 -*-
class Solution:
    def NumberOf1(self, n):
        # write code here
        count = 0
        flag = 1
        i = 0
        while i < 32 : # int 32 bit
            if n&flag != 0 :
                count += 1
            flag = flag << 1
            i += 1
        return count

注意变化处理

# -*- coding:utf-8 -*-
class Solution:
    def NumberOf1(self, n):
        # write code here
        if n < 0:
            n = n & 0xffffffff
        count = 0
        while n != 0 :
            count = count + 1
            n = n&(n-1)
        return count
