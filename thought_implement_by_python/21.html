主要是模拟数据的入栈、出栈；如果所有操作顺利完成则正确，中间有不符合的则错误

# -*- coding:utf-8 -*-
class Solution:
    def IsPopOrder(self, pushV, popV):
        # write code here
        n = len(pushV)

        fir = 0
        while fir < n and pushV[fir] != popV[0] :
            fir += 1
        if fir >= n :
            return False

        sta = pushV[0:fir]    # 当前栈存储的元素
        sta_size = fir
        pos = fir + 1    # 出栈元素的下一个位置
        i = 1
        while i < n :
            nextPopVal = popV[i]
            # 下一个出栈的元素，是上一个出栈元素的左边或者右边
            if pos < n and nextPopVal == pushV[pos] :
                pos = pos + 1
            elif sta_size != 0 and nextPopVal == sta[sta_size - 1] :
                sta_size -= 1
                sta.pop(-1)
            else :
                # 从后续元素中查找
                k = pos
                while k < n and pushV[k] != nextPopVal:
                    k += 1
                if k >= n : # 表示未找到
                    return False

                j = pos
                while j < k :
                    sta.append(pushV[j])
                    sta_size += 1
                    j += 1
                pos = k + 1

            i += 1
        return True
