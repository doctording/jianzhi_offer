非递归中序遍历

# -*- coding:utf-8 -*-
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None
class Solution:
    # 返回对应节点TreeNode
    def KthNode(self, pRoot, k):
        # write code here
        if pRoot == None or k == 0:
            return None

        sta = []
        sta.append(pRoot)
        sta_size = 1

        index = 0
        qLeft = pRoot.left
        while qLeft != None or sta_size != 0 :

            while qLeft != None :
                sta.append(qLeft)
                sta_size += 1
                qLeft = qLeft.left

            node = sta[sta_size - 1]
            sta.pop(-1)
            sta_size -= 1

            index += 1
            if index == k :
                return node

            if node.right != None :
                qLeft = node.right

        return None

递归求解

# -*- coding:utf-8 -*-
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None
class Solution:
    def __init__(self):
        self.index = 0
        self.data = None

    # 返回对应节点TreeNode
    def KthNode(self, pRoot, k):
        # write code here
        if pRoot == None or k == 0:
            return None

        def inOrder(pRoot,k) :
            if pRoot != None :
                inOrder(pRoot.left, k)
                self.index += 1
                if self.index == k :
                    self.data = pRoot
                inOrder(pRoot.right, k)

        inOrder(pRoot, k)
        return self.data
