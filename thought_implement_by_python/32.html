# -*- coding:utf-8 -*-
class Solution:
    def PrintMinNumber(self, numbers):
        # write code here
        n = len(numbers)
        if n == 0 :
            return ""
        if n == 1 :
            return numbers[0]

        li = []
        i = 0
        while i < n :
            li.append(str(numbers[i]))
            i += 1

        def comp(x, y):
            len1 = len(x)
            len2 = len(y)
            if len1 == len2 :
                if x < y:
                    return -1
                elif x > y:
                    return 1
                else:
                    return 0
            elif len1 < len2:
                i = 0
                while i < len1 :
                    if x[i] < y[i] :
                        return -1
                    elif x[i] > y[i] :
                        return 1
                    else:
                        pass
                    i += 1
                i = len1
                while i < len2:
                    if y[i] == x[len1-1] :
                        i += 1
                        continue
                    elif y[i] > x[len1-1] :
                        return -1
                    else:
                        return 1
                    i += 1
                return 0
            else :
                i = 0
                while i < len2 :
                    if x[i] < y[i] :
                        return -1
                    elif x[i] > y[i] :
                        return 1
                    else:
                        pass
                    i += 1
                i = len2
                while i < len1:
                    if x[i] == y[len2-1] :
                        i += 1
                        continue
                    elif x[i] > y[len2-1] :
                        return 1
                    else:
                        return -1
                    i += 1
                return 0

        li.sort(comp)
        strAns = ''.join(li)
        return strAns
