# -*- coding:utf-8 -*-
class Solution:
    def IsContinuous(self, numbers):
        # write code here
        n = len(numbers)
        if n != 5 :
            return False

        mp = {}
        zCnt = 0
        li2 = []
        i = 0
        while i < n :
            if numbers[i] == 0 :
                zCnt += 1
            else :
                if mp.has_key(numbers[i]) : # 不能有一样的数字
                    return False
                else :
                    li2.append(numbers[i])
                    mp[numbers[i]] = 1
            i += 1

        li2.sort() # 排序

        if zCnt == 0 :
            if li2[4] - li2[0] == 4 :
                return True
        elif zCnt == 1 :
            if li2[3] - li2[0] <= 4 and li2[3] - li2[0] >= -4:
                return True
        elif zCnt == 2 :
            if li2[2] - li2[0] <= 4 and li2[2] - li2[0] >= -4:
                return True
        elif zCnt == 3 :
            if li2[1] - li2[0] <= 4 and li2[1] - li2[0] >= -4:
                return True
        elif zCnt == 4 :
            return True
        else:
            pass

        return False
