28 数组中出现次数超过一半的数字
Table of Contents
  • 空间复杂度O(1)解法

空间复杂度O(1)解法

如果使用map，空间复杂度将是O(n)

# -*- coding:utf-8 -*-
class Solution:
    def MoreThanHalfNum_Solution(self, numbers):
        n = len(numbers)
        if n == 0 :
            return 0

        candidate = numbers[0]
        cnt = 1
        i = 1
        while i < n :
            nowNum = numbers[i]
            if cnt == 0 :
                candidate = nowNum
                cnt = 1
            else :
                if nowNum == candidate :
                    cnt += 1
                else :
                    cnt -= 1
            i += 1

        realTime = 0
        i = 0
        while i < n :
            if numbers[i] == candidate :
                realTime += 1
                if realTime > n / 2 :
                    return candidate
            i += 1

        return 0
