20 包含min函数的栈

解法1

用两个列表，一个作为栈，一个存储当前栈的min

空间复杂度:O(2n)，时间复杂度都是:O(1)

# -*- coding:utf-8 -*-
class Solution:
    def __init__(self) :
        self.data = []
        self.mdata = []
        self.cnt = 0

    def push(self, node):
        if self.cnt == 0 :
            self.mdata.append(node)
        else:
            min_val = min(node, self.mdata[self.cnt - 1])
            self.mdata.append(min_val)

        self.data.append(node)
        self.cnt += 1

    def pop(self):
        if self.cnt == 0 :
            pass
        self.data.pop(-1)
        self.mdata.pop(-1)
        self.cnt -= 1

    def top(self):
        return self.data[self.cnt - 1]

    def min(self):
        return self.mdata[self.cnt - 1]
