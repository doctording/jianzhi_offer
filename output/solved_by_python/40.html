# -*- coding:utf-8 -*-
class Solution:
    # 返回[a,b] 其中ab是出现一次的两个数字
    def FindNumsAppearOnce(self, array):
        n = len(array)
        if n <= 2 :
            return []
        rs = array[0]
        i = 1
        while i < n :
            rs = rs ^ array[i]
            i += 1

        flag = 1
        while rs & flag == 0 :
            flag = flag << 1

        n1 = 0
        n2 = 0
        i = 0
        while i < n :

            if array[i] & flag == 0 :
                n1 = n1 ^ array[i]
            else :
                n2 = n2 ^ array[i]
            i += 1

        return [n1, n2]
