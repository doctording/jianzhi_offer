快慢指针法

注意 ：

走一步，走两步；如果有环，相遇点不一定是环的入口

# -*- coding:utf-8 -*-
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None
class Solution:
    def EntryNodeOfLoop(self, pHead):
        # write code here
        if pHead == None or pHead.next == None :
            return None

        p1 = pHead # 一步
        p2 = pHead # 两步
        while p1 != None and p2 != None :

            p1 = p1.next
            if p2.next != None :
                p2 = p2.next.next
            else :
                break

            if p1 == p2 :
                # 有环, 重新开始一起走
                p1 = pHead

                while p1 != None and p2 != None :
                    if p1 == p2 :
                        return p1
                    p1 = p1.next
                    p2 = p2.next
                return None

        return None
