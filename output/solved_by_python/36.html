思路

  • 两个单链表， 有一个公共节点则必然公共节点后面的部分都是相同的(即公共尾部)

解法1

# -*- coding:utf-8 -*-
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None
class Solution:
    def FindFirstCommonNode(self, pHead1, pHead2):
        # write code here
        n1 = 0
        p1 = pHead1
        while p1 != None :
            n1 += 1
            p1 = p1.next

        n2 = 0
        p2 = pHead2
        while p2 != None :
            n2 += 1
            p2 = p2.next

        p1 = pHead1
        p2 = pHead2

        if n1 < n2 :
            i = 0
            while i < n2 - n1 :
                p2 = p2.next
                i += 1
        elif n1 > n2 :
            i = 0
            while i < n1 - n2 :
                p1 = p1.next
                i += 1
        else:
                pass

        while p1 != None and p2 != None :
            if p1 == p2 :
                return p1
            p1 = p1.next
            p2 = p2.next

        return None

解法2

借助辅助空间，少一次遍历

# -*- coding:utf-8 -*-
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None
class Solution:
    def FindFirstCommonNode(self, pHead1, pHead2):
        # write code here
        mp = {}

        p1 = pHead1
        while p1 != None :
            mp[p1] = True
            p1 = p1.next

        p2 = pHead2
        while p2 != None :
            if mp.has_key(p2) :
                return p2
            p2 = p2.next

        return None
