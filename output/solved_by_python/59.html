层次遍历

# -*- coding:utf-8 -*-
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None
class Solution:
    def Print(self, pRoot):
        # write code here
        ans = []
        if pRoot == None :
            return ans

        li = []
        li.append(pRoot)
        li_size = 1
        isLToR = True

        while li_size != 0 :
            li2 = []
            li_size2 = 0
            cengTmp = []
            i = 0
            while i < li_size :
                node = li[i]
                cengTmp.append(node.val)
                if node.left != None :
                    li2.append(node.left)
                    li_size2 += 1
                if node.right != None :
                    li2.append(node.right)
                    li_size2 += 1

                i += 1

            if isLToR :
                ans.append(cengTmp)
            else :
                cengTmp.reverse()
                ans.append(cengTmp)

            isLToR = not isLToR
            li = li2
            li_size = li_size2

        return ans

    def Print(self, pRoot):
        # write code here
        ans = []

            return ans

        li = []
        li.append(pRoot)
        li_size = 1
        isLToR = True

        while li_size != 0 :
            li2 = []

            cengTmp = []
            i = 0
            while i < li_size :
                node = li[i]
                if isLToR :
                    cengTmp.append(node.val)
                else :
                    cengTmp.insert(0, node.val)
                if node.left != None :
                    li2.append(node.left)
                    li_size2 += 1

                    li2.append(node.right)
                    li_size2 += 1

                i += 1

            ans.append(cengTmp)

            isLToR = not isLToR
            li = li2
            li_size = li_size2
